function json(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  const p      = e?.parameter || {};
  const action = p.action || '';

  if (action === 'ler')         return lerColaboradores(p.tipo || 'ativos');
  if (action === 'salvar')      return salvarColaborador(p);
  if (action === 'sincronizar') return sincronizarAba(p);

  return json({ ok:true, msg:'WeSys Template OK' });
}

// ── Leitura — retorna CSV da aba ──────────────────────────────
function lerColaboradores(tipo) {
  try {
    const sheet = getSheet(tipo);
    const data  = sheet.getDataRange().getValues();
    const csv   = data.map(row =>
      row.map(cell => {
        const s = String(cell ?? '').replace(/"/g, '""');
        return (s.includes(',') || s.includes('"') || s.includes('\n')) ? '"' + s + '"' : s;
      }).join(',')
    ).join('\n');
    return json({ ok:true, csv });
  } catch(err) { return json({ ok:false, error: err.message }); }
}

// ── Escrita — busca linha pela matrícula, não pelo número da linha ──
// ?action=salvar&tipo=ativos&matricula=WE123456&dados={"nome":"João"}
function salvarColaborador(p) {
  try {
    const tipo       = p.tipo;
    const matricula  = p.matricula || '';
    const dados      = JSON.parse(p.dados || '{}');

    if (!tipo || !matricula) throw new Error('Parâmetros incompletos');

    const colMap    = tipo === 'ativos' ? COL_ATIVOS : COL_INATIVOS;
    const colMatric = tipo === 'ativos' ? COL_ATIVOS.matricula : COL_INATIVOS.matricula; // col 5 ou 4
    const sheet     = getSheet(tipo);
    const data      = sheet.getDataRange().getValues();

    // Encontrar a linha pela matrícula (coluna correta, base 0)
    const rowIdx = data.findIndex((row, i) => i > 0 && String(row[colMatric - 1]).trim() === matricula.trim());
    if (rowIdx < 0) throw new Error('Matrícula não encontrada: ' + matricula);
    const rowNum = rowIdx + 1; // base 1 para o Sheets

    Object.entries(dados).forEach(([campo, valor]) => {
      const col = colMap[campo];
      if (col) sheet.getRange(rowNum, col).setValue(valor);
    });

    return json({ ok:true, row: rowNum });
  } catch(err) { return json({ ok:false, error: err.message }); }
}

function getSheet(tipo) {
  const gid    = tipo === 'ativos' ? GID_ATIVOS : GID_INATIVOS;
  const ss     = SpreadsheetApp.openById(SHEET_ID);
  const sheet  = ss.getSheets().find(s => s.getSheetId().toString() === gid);
  if (!sheet) throw new Error('Aba não encontrada: ' + gid);
  return sheet;
}

// doPost mantido como fallback (não funciona cross-origin, mas não quebra)
function doPost(e) { return doGet(e); }

// ── Sincronização completa — recebe linhas do banco, sobrescreve aba ──
// ?action=sincronizar&tipo=ativos&linhas=[["WeCan","SOC-RJ1",...],...]
function sincronizarAba(p) {
  try {
    const tipo   = p.tipo || 'ativos';
    const linhas = JSON.parse(p.linhas || '[]');
    if (!Array.isArray(linhas) || linhas.length === 0)
      throw new Error('Nenhuma linha recebida');

    const sheet  = getSheet(tipo);
    const ultima = sheet.getLastRow();

    // Cabeçalhos das abas (linha 1 — preservada, nunca apagada)
    const cabAtivos = [
      'BPO','C.custo','NickName','Operação','Matricula',
      'Nome completo tudo Maiúsculo Colaborador(a)',
      'Não pode ser posterior a data da atualização Data Admissão (Formato BR)',
      'Tipo de Contrato','Letra Maiuscula e sem acento Cargo (Nome completo)',
      'Salário Base (Formato BR)','Vencimento Contrato (Formato BR)',
      'Renovação (Formato BR)','Horário XX AS XX',
      'Escala Ex: 6X1 220h SEG A SAB xx:xx às xx:xx',
      'CPF','Data de Nascimento (Formato BR)','Nome Completo da Mãe',
      'Celular','E-mail','Beneficio Mobilidade','Logradouro','Numero',
      'Complemento','Bairro','Cidade','UF','CEP','Tamanho Calça',
      'Tamanho Blusa','Tamanho Calçado','Tamanho Luva',
      'Status Colaborador','Sexo','Status Operação',
      'Trintídio Mês Base','Categoria - Área'
    ];
    const cabInativos = [
      'BPO','C.custo','NickName','Matricula',
      'Nome completo tudo Maiúsculo Colaborador(a)','CPF',
      'Tipo de Contrato','Cargo (Nome completo)',
      'Data Admissão (Formato BR)','Vencimento da Demissão (Formato BR)',
      'Motivo Desligamento','Tipo do desligamento',
      'Se involuntário, por favor adicionar o nº do ticket, se houver',
      'Status Colaborador','Beneficio Mobilidade','Gênero','IDADE',
      'Categoria - Área'
    ];

    const cab = tipo === 'ativos' ? cabAtivos : cabInativos;

    // Limpar dados (linha 2 em diante), preservar cabeçalho
    if (ultima > 1) {
      sheet.getRange(2, 1, ultima - 1, cab.length).clearContent();
    }

    // Escrever cabeçalho (garante que nunca some)
    sheet.getRange(1, 1, 1, cab.length).setValues([cab]);

    // Escrever linhas de dados
    if (linhas.length > 0) {
      sheet.getRange(2, 1, linhas.length, cab.length).setValues(linhas);
    }

    return json({ ok:true, total: linhas.length });
  } catch(err) { return json({ ok:false, error: err.message }); }
}
